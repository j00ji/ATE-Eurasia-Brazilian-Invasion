            if = {
                limit = { event_target:condemned_prisoner = { religion = FROM } }
                set_character_flag = anthropo_same_religion
            }
            else = {
                set_character_flag = anthropo_different_religion
            }
            
            # Add piety based on diversity of consumption
            if = {
                limit = { has_character_flag = anthropo_different_culture }
                add_piety = 25
                if = {
                    limit = { has_character_flag = anthropo_different_religion }
                    add_piety = 25
                }
            }
            else_if = {
                limit = { has_character_flag = anthropo_same_culture }
                add_piety = 10
            }
            
            character_event = { id = modernist.403 }
        }
        
        death = {
            death_reason = death_execution_eaten
            killer = FROM
        }
    }
    
    option = {}
}

character_event = {
    id = modernist.500
    title = EVTNAME_architecturalist_holy_site
    desc = EVTDESC_architecturalist_holy_site
    picture = GFX_evt_library
    
    hide_window = yes
    is_triggered_only = yes
    
    trigger = {
        trait = architecturalist
        religion = modernist
        is_playable = yes
        prisoner = no
        NOT = { is_incapable = yes }
    }
    
    immediate = {
        # Count holy sites held
        set_variable = { which = holy_sites_held which = 0 }
        
        any_realm_title = {
            limit = {
                is_holy_site = yes
                religion = ROOT
            }
            ROOT = {
                change_variable = { which = holy_sites_held value = 1 }
            }
        }
        
        # Apply bonuses based on number of holy sites
        if = {
            limit = { check_variable = { which = holy_sites_held value = 1 } }
            character_event = { id = modernist.501 }
        }
        else_if = {
            limit = { check_variable = { which = holy_sites_held value = 2 } }
            character_event = { id = modernist.502 }
        }
        else_if = {
            limit = { check_variable = { which = holy_sites_held value = 3 } }
            character_event = { id = modernist.503 }
        }
        else_if = {
            limit = { check_variable = { which = holy_sites_held value = 4 } }
            character_event = { id = modernist.504 }
        }
        else_if = {
            limit = { check_variable = { which = holy_sites_held value = 5 } }
            character_event = { id = modernist.505 }
        }
    }
    
    option = {}
}
